Installing Sequelize
step 1
npm i sequelize --g
npm i sequelize-cli --g
npm i pg or npm install -g pg --save

step 2
creating file .sequelizerc
const path = require('path')

module.exports = {
    'config': path.resolve('./config', 'database.js'),
    'models-path': path.resolve('app', 'models'),
    'migrations-path': path.resolve('database', 'migrations'),
    'seeders-path': path.resolve('database', 'seeders'),
}
step 3
now run sequelize init
step 4
create new user model
sequelize model:generate --name User --attributes firstName:string,lastName:string,email:string,password:string
step 5 sequelize db:migrate // migrate table 
step 6 sequelize seed:generate --name user-seed
step 7 sequelize db:seed:all

options if table delete 
sequelize db:migrate:undo:all


postgresql setup in ubuntu
step 1  sudo apt update
step 2 sudo apt install postgresql
step 3 psql --version
step 4 sudo service postgresql start
step 5 sudo service postgresql status
step 6 sudo -u postgres psql login

if the connection faild then follow this steps:
sudo nano /etc/postgresql/16/main/postgresql.conf
set: listen_addresses = '*'
sudo service postgresql restart

create user only : 
sudo -u postgres psql
CREATE USER bikeserve WITH PASSWORD 'Jeevan@2026';
ALTER USER bikeserve CREATEDB;
\q for edit

check postgres status : sudo service postgresql status
check server is listen or not : sudo ss -ltnp | grep 5432
